Queues

FIFO (First-In-First-Out) data structure for managing elements

What is a Queue?

A queue is a linear data structure that follows the First-In-First-Out (FIFO) principle. Elements are added at the rear (enqueue) and removed from the front (dequeue). Think of a queue at a bank - the first person to join is the first to be served.

Key Concepts

  • Enqueue: Add an element to the rear
  • Dequeue: Remove and return the front element
  • Front: The first element in the queue
  • Rear: The last element in the queue
  • FIFO: First element added is first to be removed

Types of Queues

  • Simple Queue: Standard FIFO implementation
  • Circular Queue: Rear connects to front (efficient space usage)
  • Priority Queue: Elements served based on priority
  • Double-Ended Queue (Deque): Operations at both ends

Queue Implementation

Using Python List

class Queue:
    def __init__(self):
        self.items = []
    
    # Enqueue - O(1)
    def enqueue(self, data):
        self.items.append(data)
    
    # Dequeue - O(n) for list, O(1) for deque
    def dequeue(self):
        if not self.is_empty():
            return self.items.pop(0)
        return None
    
    # Get front element - O(1)
    def front(self):
        if not self.is_empty():
            return self.items[0]
        return None
    
    # Get rear element - O(1)
    def rear(self):
        if not self.is_empty():
            return self.items[-1]
        return None
    
    # Check if empty - O(1)
    def is_empty(self):
        return len(self.items) == 0
    
    # Get size - O(1)
    def size(self):
        return len(self.items)
    
    # Display queue
    def display(self):
        print(self.items)

Using Collections.deque (Efficient)

from collections import deque

class EfficientQueue:
    def __init__(self):
        self.items = deque()
    
    def enqueue(self, data):
        self.items.append(data)  # O(1)
    
    def dequeue(self):
        if not self.is_empty():
            return self.items.popleft()  # O(1)
        return None
    
    def front(self):
        if not self.is_empty():
            return self.items[0]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

Usage Example

# Create queue
queue = EfficientQueue()

# Enqueue elements
queue.enqueue(10)
queue.enqueue(20)
queue.enqueue(30)
queue.enqueue(40)

# Display
print(queue.items)  # Output: deque([10, 20, 30, 40])

# Get front
print(queue.front())  # Output: 10

# Dequeue
print(queue.dequeue())  # Output: 10
print(queue.dequeue())  # Output: 20

# Check size
print(queue.size())  # Output: 2

Circular Queue

class CircularQueue:
    def __init__(self, max_size):
        self.max_size = max_size
        self.items = [None] * max_size
        self.front_idx = -1
        self.rear_idx = -1
    
    def enqueue(self, data):
        if self.is_full():
            print("Queue is full!")
            return
        
        if self.front_idx == -1:
            self.front_idx = 0
        
        self.rear_idx = (self.rear_idx + 1) % self.max_size
        self.items[self.rear_idx] = data
    
    def dequeue(self):
        if self.is_empty():
            return None
        
        data = self.items[self.front_idx]
        
        if self.front_idx == self.rear_idx:
            self.front_idx = self.rear_idx = -1
        else:
            self.front_idx = (self.front_idx + 1) % self.max_size
        
        return data
    
    def is_empty(self):
        return self.front_idx == -1
    
    def is_full(self):
        return (self.rear_idx + 1) % self.max_size == self.front_idx

Complexity Analysis

Time Complexity (deque):

Enqueue: O(1) - Add to rear

Dequeue: O(1) - Remove from front

Front/Rear: O(1) - View element

Search: O(n) - May check all elements

Space Complexity:

O(n) - Where n is number of elements

Applications

  • CPU Scheduling: Process scheduling in operating systems
  • Printer Queue: Managing print jobs
  • BFS: Breadth-first search in graphs
  • Level-order Traversal: Tree traversal
  • Message Queue: Asynchronous message processing
  • Load Balancing: Distributing tasks evenly

1. Implement Queue Using Stacks

class QueueUsingStacks:
    def __init__(self):
        self.stack1 = []  # For enqueue
        self.stack2 = []  # For dequeue
    
    def enqueue(self, data):
        self.stack1.append(data)  # O(1)
    
    def dequeue(self):
        if not self.stack2:
            if not self.stack1:
                return None
            while self.stack1:
                self.stack2.append(self.stack1.pop())
        
        return self.stack2.pop() if self.stack2 else None  # O(1) amortized

2. Implement Priority Queue

import heapq

class PriorityQueue:
    def __init__(self):
        self.items = []
    
    # Enqueue with priority (lower = higher priority)
    def enqueue(self, data, priority):
        heapq.heappush(self.items, (priority, data))
    
    # Dequeue highest priority element
    def dequeue(self):
        if not self.is_empty():
            return heapq.heappop(self.items)[1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

3. Implement Deque (Double-Ended Queue)

class Deque:
    def __init__(self):
        self.items = deque()
    
    def add_front(self, data):
        self.items.appendleft(data)
    
    def add_rear(self, data):
        self.items.append(data)
    
    def remove_front(self):
        if not self.is_empty():
            return self.items.popleft()
        return None
    
    def remove_rear(self):
        if not self.is_empty():
            return self.items.pop()
        return None
    
    def is_empty(self):
        return len(self.items) == 0

4. Sliding Window Maximum

def sliding_window_maximum(arr, k):
    """
    Find maximum for each sliding window of size k.
    Time Complexity: O(n)
    Space Complexity: O(k)
    """
    from collections import deque
    
    dq = deque()  # Store indices
    result = []
    
    for i in range(len(arr)):
        # Remove elements outside current window
        if dq and dq[0] < i - k + 1:
            dq.popleft()
        
        # Remove smaller elements
        while dq and arr[dq[-1]] < arr[i]:
            dq.pop()
        
        dq.append(i)
        
        # Add to result
        if i >= k - 1:
            result.append(arr[dq[0]])
    
    return result

# Test
print(sliding_window_maximum([1, 3, 1, 2, 0, 5], 3))
# Output: [3, 3, 2, 5]
